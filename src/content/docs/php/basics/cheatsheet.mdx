```js
  // Array creation
  const array1 = [1, 2, 3, 4];
  const array2 = new Array(1, 2, 3, 4);
  
  // Adding/removing elements
  array1.push(5);       // Add to end: [1, 2, 3, 4, 5]
  array1.pop();         // Remove from end: [1, 2, 3, 4]
  array1.unshift(0);    // Add to beginning: [0, 1, 2, 3, 4]
  array1.shift();       // Remove from beginning: [1, 2, 3, 4]
  
  // Finding elements
  array1.indexOf(3);    // Returns 2
  array1.includes(5);   // Returns false
```
